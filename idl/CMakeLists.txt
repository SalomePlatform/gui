INCLUDE(UseOmniORB)  # Provided by KERNEL

  ${OMNIORB_INCLUDE_DIR}
)

  SalomeApp_Engine.idl
)

  ${CMAKE_CURRENT_SOURCE_DIR}
  ${KERNEL_ROOT_DIR}/idl/salome
)

  ${KERNEL_SalomeIDLKernel}
)
